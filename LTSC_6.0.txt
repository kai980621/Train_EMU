        function updatePhysics() {
            const gear = CONFIG.GEARS[state.currentGear];
            let delta = (gear.accel || 0) - (gear.decel || 0);

            // Industrial Physics Refinement
            if (state.currentGear === 5) {
                delta = -0.08; // Air resistance only
            } else if (state.currentGear < 5) {
                // Braking is more effective at higher speeds
                delta *= (1 + state.currentSpeed / 150);

            state.currentSpeed += delta * (1.0 / CONFIG.PHYSICS_HZ);
            if (state.currentSpeed < 0.05) state.currentSpeed = 0;
            state.currentSpeed = Math.min(CONFIG.MAX_SPEED, state.currentSpeed);

                if (state.doorsOpen) {
                    rate = 1.0;
                    if (vPlayer.getPlayerState() !== 1) { vPlayer.playVideo(); aPlayer.playVideo(); }

                    const elapsed = Date.now() - (state.doorCloseTime || 0);

                    if (state.currentSpeed <= 0) {
                        // 加速前或完全停止時

                        }
                    } else {
                        // 移動中
                        if (vPlayer.getPlayerState() !== 1 && !state.isPaused) { vPlayer.playVideo(); aPlayer.playVideo(); }

                        // 關門起步 15 秒內，且車速尚低時，強制 1.0 倍速 (正常播)
                        if (elapsed < 15000 && state.currentSpeed < 15) {
                            rate = 1.0;
                        } else {
                            // 其他情況根據車速，最低 0.25 倍速
                            rate = Math.max(0.25, state.currentSpeed / 60.0);
                        }
                    }
                }

                rate = Math.max(0.25, Math.min(2.0, rate));
                if (Math.abs(rate - state.lastRate) > 0.05) {
                    vPlayer.setPlaybackRate(rate);
                    state.lastRate = rate;

            }
        }

            state.doorsOpen = true;
            state.doorCloseTime = 0;
            sfx.doorIn.play().catch(() => { });
            const hud = $('door-hud');
            hud.className = 'glass-card door-open';
            hud.textContent = '車門開啟';
            if (state.currentGear > 5) applyGear(5);
            vPlayer.playVideo(); aPlayer.playVideo();

            // Manual Station Sync & Discovery
            const stations = state.memory[state.videoId] || [];
            const idx = stations.findIndex(s => Math.abs(s.time - curTime) < 60);

            if (idx !== -1) {
                const stn = stations[idx];
                const diff = Math.abs(curTime - stn.time);
                if (state.currentSpeed === 0 && !state.evalShown) {
                    showEvaluation(diff);
                    state.evalShown = true;
                }
                // Automatic Data Update
                stn.time = curTime;
                state.targetIndex = idx;
                saveMemory();
            } else if (!stations.find(s => Math.abs(s.time - curTime) < 40)) {
                $('smart-discovery').style.display = 'block';
                $('new-stn-name').value = "";
                $('new-stn-name').focus();
            }
        }


        function closeDoors() {
            const curTime = vPlayer.getCurrentTime();
            state.doorsOpen = false;
            state.doorCloseTime = Date.now();
            sfx.doorOut.play().catch(() => { });
            const hud = $('door-hud');
            hud.className = 'glass-card door-closed';
            hud.textContent = '車門鎖定';

            const stations = state.memory[state.videoId] || [];
            const idx = stations.findIndex(s => Math.abs(s.time - curTime) < 60);
            if (idx !== -1) {
                const stn = stations[idx];
                stn.closeTime = curTime;
                // Move to next station target
                state.targetIndex = Math.min(stations.length - 1, idx + 1);
                saveMemory();
            }
        }


        function saveMemory() {
            localStorage.setItem('edgerail_st_v5', JSON.stringify(state.memory));
            refreshStationList();
        }

        // Modal Helpers
        function toggleModal(id) {
            const m = $(id);
            const isVisible = m.style.display === 'block';
            hideModals();
            if (!isVisible) m.style.display = 'block';
        }

            } else {

            }
        };
        $('btn-quick-sync').onclick = () => handleAction('Sync');
        $('btn-pause-toggle').onclick = () => {

        function updateTimeline(curTime, stations) {
            const container = $('route-timeline');
            if (stations.length === 0) { container.style.display = 'none'; return; }
            container.style.display = 'block';

            const lastStn = stations[stations.length - 1];
            const totalDuration = lastStn.time + 60;
            container.innerHTML = stations.map((s, i) => {
                const pos = (s.time / totalDuration) * 100;
                let cls = 'timeline-node';
                if (i === state.targetIndex) cls += ' active';
                else if (curTime > s.time) cls += ' passed';
                return `<div class="${cls}" style="top: ${pos}%"></div>`;
            }).join('');
        }

        // Station Management
        function refreshStationList() {
            const list = $('stn-list-scroll');
            const data = state.memory[state.videoId] || [];
            if (data.length === 0) {
                list.innerHTML = '<div style="opacity:0.3;text-align:center;padding:20px;">NO ROUTE DATA IN MEMORY</div>';
                return;
            }
            list.innerHTML = data.map((s, i) => `
                <div class="stn-item" style="display: flex; flex-direction: column; align-items: stretch; gap: 8px;">
                    <div style="display: flex; gap: 10px; align-items: center;">
                        <input type="text" value="${s.name}" onchange="updateStn(${i}, 'name', this.value)" style="flex:1; font-weight:900; background:rgba(255,255,255,0.05); text-align:left;">
                        <button class="action-btn btn-red" onclick="removeStn(${i})" style="width:32px; height:32px; padding:0; border-radius:50%; margin:0;">×</button>
                    </div>
                    <div style="display: grid; grid-template-columns: 1fr 1fr; gap: 8px; font-size: 11px;">
                        <div>
                            <span style="opacity:0.4; font-size:9px;">ARRIVE (S)</span>
                            <input type="number" step="0.1" value="${s.time}" onchange="updateStn(${i}, 'time', this.value)" style="height:30px; padding:4px;">
                        </div>
                        <div>
                            <span style="opacity:0.4; font-size:9px;">DEPART (S)</span>
                            <input type="number" step="0.1" value="${s.closeTime || 0}" onchange="updateStn(${i}, 'closeTime', this.value)" style="height:30px; padding:4px;">
                        </div>
                    </div>
                </div>
            `).join('');
        }

        function formatTime(s) {
            const m = Math.floor(s / 60);
            const sec = Math.floor(s % 60);
            return `${m}:${sec.toString().padStart(2, '0')}`;
        }

        window.updateStn = (i, key, val) => {
            const numVal = (key === 'name') ? val : parseFloat(val);
            state.memory[state.videoId][i][key] = numVal;
            if (key === 'time') state.memory[state.videoId].sort((a, b) => a.time - b.time);
            saveMemory();
        };

        window.removeStn = (i) => {
            state.memory[state.videoId].splice(i, 1);
            saveMemory();
        };

        window.memorizeLocation = () => {
            const name = $('new-stn-name').value.trim() || '未命名站點';
            const time = vPlayer.getCurrentTime();
            if (!state.memory[state.videoId]) state.memory[state.videoId] = [];
            state.memory[state.videoId].push({ name, time, closeTime: time + 20 });
            state.memory[state.videoId].sort((a, b) => a.time - b.time);
            saveMemory();
            $('smart-discovery').style.display = 'none';
        }

        // Bulk IO
        window.importData = () => {
            try {
                let raw = $('bulk-io').value.trim();
                if (!raw) return alert("輸入區為空");

                if (raw.startsWith('http')) {
                    const m = raw.match(/(?:v=|\/)([0-9A-Za-z_-]{11})/);
                    if (m) { $('yt-url-input').value = raw; $('btn-fetch-yt').click(); return; }
                }

                let data;
                try {
                    data = JSON.parse(raw);
                } catch (e) {
                    // Try parsing as the custom text format: Name Arrival Departure
                    const lines = raw.split('\n').filter(l => l.trim());
                    data = lines.map(line => {
                        const parts = line.trim().split(/\s+/);
                        if (parts.length < 3) return null;

                        const departure = parts.pop();
                        const arrival = parts.pop();
                        const name = parts.join(' ');

                        const t2s = (t) => {
                            if (!t || t === '---') return null;
                            const p = t.split(':');
                            if (p.length === 2) return parseInt(p[0]) * 60 + parseFloat(p[1]);
                            return parseFloat(t);
                        };

                        const arr = t2s(arrival);
                        const dep = t2s(departure);

                        return {
                            name,
                            time: arr === null ? (dep !== null ? 0 : 0) : arr,
                            closeTime: dep === null ? 0 : dep
                        };
                    }).filter(x => x);

                    if (data.length === 0) throw new Error("Format error");
                }

                if (Array.isArray(data)) {
                    if (!state.videoId) return alert("匯入站點前請先連結影像");
                    state.memory[state.videoId] = data.map(s => ({
                        name: s.name || '未命名站點',
                        time: parseFloat(s.time) || 0,
                        closeTime: parseFloat(s.closeTime) || 0
                    })).sort((a, b) => a.time - b.time);
                    saveMemory();
                    alert(`成功匯入 ${state.memory[state.videoId].length} 個站點`);
                } else if (typeof data === 'object') {
                    Object.assign(state.memory, data);
                    saveMemory();
                    alert("工作環境已完全恢復");
                }
            } catch (e) {
                console.error(e);
                alert("匯入失敗：格式不支援或資料損壞");
            }
        }

        window.exportDataToClipboard = () => {
            const data = state.memory[state.videoId] || [];
            const str = JSON.stringify(data, null, 2);
            navigator.clipboard.writeText(str);
            $('bulk-io').value = str;
        }

        window.downloadData = () => {
            const data = state.memory[state.videoId] || [];
            if (data.length === 0) return alert("無資料可供匯出");
            const blob = new Blob([JSON.stringify(data, null, 2)], { type: 'application/json' });
            const url = URL.createObjectURL(blob);
            const a = document.createElement('a');
            a.href = url;
            a.download = `route_${state.videoId}_${Date.now()}.json`;
            a.click();
            URL.revokeObjectURL(url);
        }

        window.clearFleetData = () => {
            if (confirm("警告：確定要清除此影片的所有站點資料？此操作無法還原。")) {
                state.memory[state.videoId] = [];
                saveMemory();
            }
        }

        // Video Switching
        let pendingVid = null;
        $('btn-fetch-yt').onclick = async () => {
            const url = $('yt-url-input').value;
            const m = url.match(/(?:v=|\/)([0-9A-Za-z_-]{11})/);
            if (!m) return alert("無效的網址");

            $('btn-fetch-yt').textContent = '獲取中...';
            try {
                const res = await fetch(`https://www.youtube.com/oembed?url=https://www.youtube.com/watch?v=${m[1]}&format=json`);
                const data = await res.json();
                pendingVid = m[1];
                $('yt-preview-title').textContent = data.title;
                $('yt-preview').style.display = 'block';
                $('yt-apply-group').style.display = 'grid';
                $('btn-fetch-yt').style.display = 'none';
            } catch (e) {
                alert("獲取失敗");
                $('btn-fetch-yt').textContent = '讀取資料';
            }
        };

        $('btn-cancel-yt').onclick = () => {
            pendingVid = null;
            $('yt-preview').style.display = 'none';
            $('yt-apply-group').style.display = 'none';
            $('btn-fetch-yt').style.display = 'block';
            $('btn-fetch-yt').textContent = '讀取資料';
        };

        $('btn-apply-yt').onclick = () => {
            state.videoId = pendingVid;
            vPlayer.loadVideoById(state.videoId);
            aPlayer.loadVideoById(state.videoId);
            state.currentSpeed = 0;
            state.doorsOpen = true;
            openDoors();
            hideModals();

            // Re-check Cloud Library
            setTimeout(() => checkCloudLibrary(state.videoId), 2000);

            refreshStationList();
        };

        // Evaluation
        function showEvaluation(diff) {
            state.evalShown = true;
            const overlay = $('eval-overlay');
            const grade = overlay.querySelector('.eval-grade');
            const dist = (diff * 16.6).toFixed(1);

            let text = '不佳', color = 'var(--danger)', comment = "";
            if (diff < 0.3) {
                text = '完美'; color = 'gold';
                comment = "極致的控制力！誤差不到 30 釐米，乘客完全感覺不到晃動。";
            }
            else if (diff < 1.0) {
                text = '優秀'; color = 'var(--success)';
                comment = "相當準確的停靠。繼續保持這種穩定的煞車曲線。";
            }
            else if (diff < 3.0) {
                text = '良好'; color = 'var(--accent)';
                comment = "安全停靠，但仍有進步空間。建議提早進入 B1 階段。";
            } else {
                comment = "停車誤差較大。請注意 TASC 系統的建議減速點。";
            }

            grade.textContent = text; grade.style.color = color;
            $('eval-detail').textContent = `誤差: ${dist}m (${diff.toFixed(2)}s)`;
            overlay.style.opacity = '1';
            setTimeout(() => { overlay.style.opacity = '0'; }, 3500);
        }

        // Jump Menu
        window.openJumpMenu = (dir) => {
            const time = vPlayer.getCurrentTime();
            const stations = state.memory[state.videoId] || [];
            let filtered = [];

            if (dir > 0) {
                $('jump-title').textContent = '下一站';
                filtered = stations.filter(s => s.time > time + 5);
            } else {
                $('jump-title').textContent = '上一站';
                filtered = [...stations].filter(s => s.time < time - 5).reverse();
            }

            const list = $('jump-list');
            if (filtered.length === 0) list.innerHTML = '<p style="opacity:0.3;text-align:center;">空</p>';
            else {
                list.innerHTML = filtered.map(s => `
                    <button class="action-btn" onclick="executeJump(${s.time})">
                        ${s.name} <span style="float:right;opacity:0.5;">${formatTime(s.time)}</span>
                    </button>
                `).join('');
            }
            toggleModal('jump-picker');
        }

        window.executeJump = (t) => {
            vPlayer.seekTo(t, true);
            aPlayer.seekTo(t, true);
            state.currentSpeed = 0;
            state.doorsOpen = true;
            openDoors();
            hideModals();
        }

        // Touch Interaction
        const track = $('mascon-track');
        const handleDrag = (e) => {
            const rect = track.getBoundingClientRect();
            const y = (e.touches ? e.touches[0].clientY : e.clientY) - rect.top;
            const percent = Math.max(0, Math.min(1, y / rect.height));
            applyGear(10 - Math.round(percent * 10));
        };

        $('mascon-handle').addEventListener('touchstart', (e) => {
            window.addEventListener('touchmove', handleDrag);
            window.addEventListener('touchend', () => window.removeEventListener('touchmove', handleDrag), { once: true });
        });
        $('mascon-handle').addEventListener('mousedown', (e) => {
            window.addEventListener('mousemove', handleDrag);
            window.addEventListener('mouseup', () => window.removeEventListener('mousemove', handleDrag), { once: true });
        });

        // Key Listeners
        window.addEventListener('keydown', (e) => {
            if (document.activeElement.tagName === 'INPUT' || document.activeElement.tagName === 'TEXTAREA') return;
            handleAction(e.code);
        });
