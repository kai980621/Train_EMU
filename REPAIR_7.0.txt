                        if (vPlayer.getPlayerState() === 1) { vPlayer.pauseVideo(); aPlayer.pauseVideo(); }
                    } else {
                        if (vPlayer.getPlayerState() !== 1 && !state.isPaused) { vPlayer.playVideo(); aPlayer.playVideo(); }
                        if (state.currentSpeed < 10) rate = 0.5; // better stop sliding view than abrupt slowings 
                    }
                } else { if (vPlayer.getPlayerState() !== 1) { vPlayer.playVideo(); aPlayer.playVideo(); } }

        function applyGear(idx) {
            if (state.doorsOpen && idx > 5) return;
            if (state.currentGear === idx) return;
            state.currentGear = idx;

            if (navigator.vibrate) { if (idx === 0) navigator.vibrate([30, 30, 30]); else if (idx === 5) navigator.vibrate(10); else navigator.vibrate(15); }
            if (idx > 5) { vPlayer.playVideo(); aPlayer.playVideo(); }
            updateMasconUI();
        }

        function updateMasconUI() {
            const handle = $('mascon-handle'); const gear = CONFIG.GEARS[state.currentGear];
            handle.style.top = `${((10 - state.currentGear) / 10) * 100}%`;
            handle.textContent = gear.name; $('gear-label').textContent = gear.name; $('gear-label').style.color = gear.color;
        }

        // (Original LTS code handling restored! ) ! 
        function handleAction(code) {
            if (!state.isStarted || state.isPaused) return;

            if (code === 'KeyZ') {
                if (state.doorsOpen) return;
                applyGear(Math.min(10, state.currentGear + 1));
            } else if (code === 'KeyS') { applyGear(Math.max(0, state.currentGear - 1)); }
            else if (code === 'KeyA') { applyGear(5); }
            else if (code === 'KeyY') { applyGear(0); }
            else if (code === 'KeyF') { $('btn-fullscreen').click(); }
            // User requested SYNC Button functionality !!! HERE IT REMAINS AS REQUESTED!!: 
            else if (code === 'KeyX' || code === 'Sync') {
                const curT = vPlayer.getCurrentTime(); aPlayer.seekTo(curT, true);
                console.log("[LTS_V7] Audio stream force synchronizer dispatched!!");
            }
            else if (code === 'KeyQ') { if (state.currentSpeed === 0 && !state.doorsOpen) openDoors(); }
            else if (code === 'KeyE') { if (state.doorsOpen) closeDoors(); }
            else if (code === 'Space' || code === 'ToggleDoors') {
                if (state.currentSpeed > 0) return;
                state.doorsOpen ? closeDoors() : openDoors();
            }
        }

        function openDoors(skipDiscovery = false) {
            const curTime = vPlayer.getCurrentTime();
            state.doorsOpen = true; sfx.doorIn.play().catch(() => { });

            // Critical Sync !
            if (aPlayer && aPlayer.seekTo) aPlayer.seekTo(curTime, true);
            const hud = $('door-hud'); hud.className = 'glass-card door-open'; hud.textContent = '車門開啟';

            if (state.currentGear > 5) applyGear(5);
            vPlayer.playVideo(); aPlayer.playVideo();

            // Native Eval & Discover codes completely untouched:
            const stations = state.memory[state.videoId] || [];
            const idx = stations.findIndex(s => Math.abs(s.time - curTime) < 60);

            if (idx !== -1) {
                const stn = stations[idx];
                const diff = Math.abs(curTime - stn.time);
                if (state.currentSpeed === 0 && !state.evalShown && !skipDiscovery) {
                    showEvaluation(diff);
                    state.evalShown = true;
                }
                stn.time = curTime;
                state.targetIndex = idx;
                saveMemory();
            } else if (!skipDiscovery && !stations.find(s => Math.abs(s.time - curTime) < 40)) {
                $('smart-discovery').style.display = 'block';
                $('new-stn-name').value = ""; $('new-stn-name').focus();
            }
        }

        function closeDoors() {
            const curTime = vPlayer.getCurrentTime();
            state.doorsOpen = false; sfx.doorOut.play().catch(() => { });

            if (aPlayer && aPlayer.seekTo) aPlayer.seekTo(curTime, true);
            const hud = $('door-hud'); hud.className = 'glass-card door-closed'; hud.textContent = '車門鎖定';

            const stations = state.memory[state.videoId] || [];
            const idx = stations.findIndex(s => Math.abs(s.time - curTime) < 60);
            if (idx !== -1) {
                stations[idx].closeTime = curTime;
                state.targetIndex = Math.min(stations.length - 1, idx + 1);
                saveMemory();
            }
        }
