    <script>
        // === Core Engine State ===
        let visualPlayer = null;
        let audioPlayer = null;
        let apiReady = false;
        let isStarted = false;
        let isPaused = false;

        let currentVideoId = 'A0Oub6N2VqM'; 

        // === Sync Engine ===
        let syncCheckInterval = null;

        // 重新恢復「真正操控播放速率」的真實駕駛感
        // 將不同檔位嚴格對應 YouTube 容許的倍速，帶來純粹的加速、減速回饋！
        // 同時修復斷音問題，秘訣在於：只有畫面(Visual)會變速，音效(Audio)始終維持在 1.0 播報播音！
        const GEARS = [
            { id: 0, name: 'EB', rate: 0.0,  class: 'gear-EB' },
            { id: 1, name: 'B4', rate: 0.25, class: 'gear-B' },
            { id: 2, name: 'B3', rate: 0.5,  class: 'gear-B' },
            { id: 3, name: 'B2', rate: 0.75, class: 'gear-B' },
            { id: 4, name: 'B1', rate: 0.85, class: 'gear-B' },
            { id: 5, name: 'N',  rate: 1.0,  class: 'gear-N' },
            { id: 6, name: 'P1', rate: 1.25, class: 'gear-P' },
            { id: 7, name: 'P2', rate: 1.5,  class: 'gear-P' },
            { id: 8, name: 'P3', rate: 2.0,  class: 'gear-P' },
            { id: 9, name: 'P4', rate: 2.5,  class: 'gear-P' },
            { id: 10, name: 'P5', rate: 3.0, class: 'gear-P' }
        ];
        
        let currentGearId = 0; // Starts with EB (0.0x / Paused)

        // 停車判定系統相關狀態
        let brakeScore = 100;
        let isStoppedState = true; 
        
        // DOM Elements
        const btnMasterStart = document.getElementById('btn-master-start');
        const startupScreen = document.getElementById('startup-screen');
        const appContainer = document.getElementById('app-container');
        const handle = document.getElementById('mascon-handle');
        const track = document.getElementById('mascon-track');
        const gearDisplay = document.getElementById('gear-display');
        const speedValue = document.getElementById('speed-value');

        // === YouTube API Initialization ===
        function onYouTubeIframeAPIReady() {
            apiReady = true;
            btnMasterStart.textContent = '啟動駕駛台 (START)';
            btnMasterStart.disabled = false;
        }

        function createPlayers() {
            const playerVars = {
                'autoplay': 1, 'controls': 0, 'disablekb': 1, 'fs': 0, 
                'modestbranding': 1, 'rel': 0, 'showinfo': 0, 'iv_load_policy': 3,
                'playsinline': 1, 'enablejsapi': 1, 'origin': window.location.origin
            };
            const host = 'https://www.youtube-nocookie.com';

            visualPlayer = new YT.Player('visual-player-container', {
                videoId: currentVideoId, playerVars: playerVars, host: host,
                events: { 'onReady': onVisualPlayerReady, 'onStateChange': onPlayerStateChange }
            });

            audioPlayer = new YT.Player('audio-player-container', {
                videoId: currentVideoId, playerVars: playerVars, host: host,
                events: { 'onReady': onAudioPlayerReady }
            });
        }

        let visualReady = false; let audioReady = false;
        function checkBothReady() {
            if (visualReady && audioReady && !isStarted) {
                visualPlayer.mute(); 
                audioPlayer.unMute();
                audioPlayer.setVolume(100);

                startupScreen.style.display = 'none';
                appContainer.style.display = 'block';
                isStarted = true;

                applyGearEffect(currentGearId);
                startSyncEngine();
                updateSpeedometer();
            }
        }

        function onVisualPlayerReady(event) { visualReady = true; checkBothReady(); }
        function onAudioPlayerReady(event) { audioReady = true; checkBothReady(); }

        function onPlayerStateChange(event) {
            if (event.data === YT.PlayerState.ENDED) {
                visualPlayer.seekTo(0); audioPlayer.seekTo(0);
            }
            if (event.data === YT.PlayerState.PLAYING && visualPlayer.setPlaybackQuality) {
                visualPlayer.setPlaybackQuality('hd1080');
            }
        }

        btnMasterStart.addEventListener('click', () => {
            btnMasterStart.textContent = '連線中...';
            btnMasterStart.disabled = true;
            createPlayers();
            setTimeout(() => { document.getElementById('btn-unmute').style.display = 'block'; }, 4000);
        });

        function startSyncEngine() {
            if (syncCheckInterval) clearInterval(syncCheckInterval);
            syncCheckInterval = setInterval(() => {
                if (!isStarted || isPaused || isSeeking) return;

                if (visualPlayer.getPlayerState && audioPlayer.getPlayerState) {
                    const vState = visualPlayer.getPlayerState();
                    const aState = audioPlayer.getPlayerState();

                    // 確保影片沒有被 YT 的 buffering 循環卡住
                    if (currentGearId > 0 && (vState === YT.PlayerState.PAUSED || vState === YT.PlayerState.UNSTARTED)) {
                        visualPlayer.playVideo();
                    }
                    if (currentGearId > 0 && (aState === YT.PlayerState.PAUSED || aState === YT.PlayerState.UNSTARTED)) {
                        audioPlayer.playVideo();
                    }
                    
                    // 當在 N 檔時 (1.0x)，避免音頻輕微漂移
                    if (currentGearId === 5 && vState === YT.PlayerState.PLAYING) {
                        const vpTime = visualPlayer.getCurrentTime();
                        const apTime = audioPlayer.getCurrentTime();
                        if (Math.abs(vpTime - apTime) > 1.5) {
                            audioPlayer.seekTo(vpTime, true);
                        }
                    }
                }
                updateSpeedometer();
            }, 200);
        }

        // === Speedometer ===
        let displaySpeed = 0;
        function updateSpeedometer() {
            if (isPaused) { speedValue.textContent = "0"; return; }
            
            // 透過實際播放速率換算出現在的視覺速度
            const targetRate = GEARS[currentGearId].rate;
            const targetSpeed = targetRate * 85;

            // 平滑時速轉換，煞車(B檔)減速曲線更明顯
            let approachRatio = 0.05;
            if (currentGearId === 0) approachRatio = 0.15; 
            if (currentGearId > 5) approachRatio = 0.02;

            displaySpeed += (targetSpeed - displaySpeed) * approachRatio;

            // 完美的 0 km/h 停止並觸發【停車判定】
            if (currentGearId === 0 && displaySpeed < 1.0) {
                if (displaySpeed > 0 && !isStoppedState) {
                    triggerStopEvaluation();
                }
                displaySpeed = 0;
            }

            speedValue.textContent = Math.round(displaySpeed);
        }

        // 停車判定結算系統
        function triggerStopEvaluation() {
            isStoppedState = true;
            const ratingPanel = document.getElementById('stop-rating');
            
            let message = "停車判定：良好 Good";
            let color = "#00ffcc";

            if (brakeScore >= 90) {
                message = "停車判定：完美 Perfect!";
                color = "gold";
            } else if (brakeScore <= 50) {
                message = "停車判定：急煞扣分 Bad";
                color = "#ff3333";
            }

            ratingPanel.textContent = message;
            ratingPanel.style.color = color;
            ratingPanel.style.textShadow = `0 0 20px ${color}`;
            
            ratingPanel.style.display = 'block';
            ratingPanel.classList.remove('rating-anim');
            void ratingPanel.offsetWidth; // trigger reflow
            ratingPanel.classList.add('rating-anim');
        }


        // === Mascon Interaction ===
        let isDragging = false;
        function renderGearLabels() {
            const container = document.getElementById('gear-labels');
            container.innerHTML = '';
            [...GEARS].reverse().forEach((gear, idx) => {
                const label = document.createElement('div');
                label.className = 'gear-label';
                label.textContent = gear.name;
                if (gear.name === 'N') { label.style.color = '#fff'; label.style.fontSize = '14px'; }
                if (gear.name === 'EB') label.style.color = '#ff3333';
                label.style.position = 'absolute';
                label.style.top = `${(idx / (GEARS.length - 1)) * 100}%`;
                container.appendChild(label);
            });
        }
        renderGearLabels();

        function updateMasconUI() {
            const gear = GEARS[currentGearId];
            const maxIdx = GEARS.length - 1;
            const percentage = ((maxIdx - currentGearId) / maxIdx) * 100;

            handle.style.top = `${percentage}%`;
            handle.textContent = gear.name;
            handle.style.borderColor = gear.name === 'EB' ? '#ff3333' : '#fff';
            gearDisplay.textContent = gear.name;
            gearDisplay.className = `gear-text-indicator ${gear.class}`;
        }

        function applyGearEffect(gearId) {
            let previousGear = currentGearId;
            currentGearId = gearId;
            
            // 【評分邏輯紀錄】: 監測有無「退檔過快」的行為造成急煞
            if (gearId < previousGear) {
                let dropAmount = previousGear - gearId;
                if (dropAmount >= 4) {
                    brakeScore -= 30; // 瞬間降太多檔位大扣分
                } else if (dropAmount >= 2) {
                    brakeScore -= 10;
                }
            }

            // 【重置狀態】: 只要往前推到 P 以上，就算重新發車
            if (gearId > 0 && isStoppedState) {
                isStoppedState = false;
                brakeScore = 100;
                document.getElementById('stop-rating').style.display = 'none';
            }

            updateMasconUI();

            if (!visualPlayer || !audioPlayer) return;
            const gear = GEARS[gearId];

            if (gear.id === 0) { // EB - Video pauses
                visualPlayer.pauseVideo();
                // We leave Audio playing its normal 1.0x course to not cut off announcements!
            } else {
                if (visualPlayer.getPlayerState() !== YT.PlayerState.PLAYING && visualPlayer.getPlayerState() !== YT.PlayerState.BUFFERING) {
                    visualPlayer.playVideo();
                }
                if (audioPlayer.getPlayerState() !== YT.PlayerState.PLAYING && audioPlayer.getPlayerState() !== YT.PlayerState.BUFFERING) {
                    audioPlayer.playVideo();
                }
                
                try {
                    // 真實減速 / 加速的關鍵！
                    visualPlayer.setPlaybackRate(gear.rate);
                } catch (e) { console.log(e); }
            }

            // 【重新對齊音效】
            // 由於煞車時(B1-B4)我們的視覺播放速度在0.2~0.8，但音訊會保持 1.0 (為了聲音不走調、不斷訊)
            // 所以只要我們回到 N 檔 (1.0x)，我們就幫它們做一次性同步，確保後面的畫面完美對上！
            if (gear.id === 5) {
                const vt = visualPlayer.getCurrentTime();
                audioPlayer.seekTo(vt, true);
            }
        }

        // Pointer Events Setup
        track.addEventListener('pointerdown', handlePointerDown);
        window.addEventListener('pointermove', handlePointerMove);
        window.addEventListener('pointerup', handlePointerUp);

        function handlePointerDown(e) { isDragging = true; handlePointerMove(e); }
        function handlePointerMove(e) {
            if (!isDragging) return;
            e.preventDefault();
            const rect = track.getBoundingClientRect();
            let percentage = Math.max(0, Math.min(1, (e.clientY - rect.top) / rect.height));
            let newGearIdx = (GEARS.length - 1) - Math.round(percentage * (GEARS.length - 1));

            if (newGearIdx !== currentGearId) {
                if (navigator.vibrate) navigator.vibrate(20);
                applyGearEffect(newGearIdx);
            }
        }
        function handlePointerUp() { isDragging = false; }

        // === Keyboard Support ===
        window.addEventListener('keydown', (e) => {
            if (isSeeking || isPaused || !isStarted) return;
            if (e.target.tagName.toLowerCase() === 'input') return;

            let newIdx = currentGearId;
            if (e.key === 'ArrowUp' || e.key === 'w' || e.key === 'W') {
                newIdx = Math.min(newIdx + 1, GEARS.length - 1); 
            } else if (e.key === 'ArrowDown' || e.key === 's' || e.key === 'S') {
                newIdx = Math.max(newIdx - 1, 0); 
            }
            if (newIdx !== currentGearId) applyGearEffect(newIdx);
        });

        // === Side Panel Controls ===
        document.getElementById('btn-fullscreen').addEventListener('click', () => {
            if (!document.fullscreenElement) {
                document.documentElement.requestFullscreen().catch(err => { alert("無法切換全螢幕"); });
            } else {
                if (document.exitFullscreen) document.exitFullscreen();
            }
        });

        document.getElementById('btn-seek-bw').addEventListener('click', () => {
            if (!visualPlayer) return;
            isSeeking = true;
            const target = Math.max(0, visualPlayer.getCurrentTime() - 60);
            visualPlayer.seekTo(target, true); audioPlayer.seekTo(target, true);
            setTimeout(() => isSeeking = false, 500);
        });

        document.getElementById('btn-seek-fw').addEventListener('click', () => {
            if (!visualPlayer) return;
            isSeeking = true;
            const target = visualPlayer.getCurrentTime() + 60;
            visualPlayer.seekTo(target, true); audioPlayer.seekTo(target, true);
            setTimeout(() => isSeeking = false, 500);
        });

        document.getElementById('btn-pause').addEventListener('click', () => {
            if (!visualPlayer) return;
            isPaused = !isPaused;
            if (isPaused) {
                visualPlayer.pauseVideo(); audioPlayer.pauseVideo();
                document.getElementById('btn-pause').style.background = 'rgba(255,100,100,0.8)';
            } else {
                visualPlayer.playVideo(); audioPlayer.playVideo();
                document.getElementById('btn-pause').style.background = 'rgba(0,0,0,0.6)';
                audioPlayer.seekTo(visualPlayer.getCurrentTime(), true);
            }
        });

        document.getElementById('btn-change-video').addEventListener('click', () => {
            const sel = document.getElementById('video-selector');
            sel.style.display = sel.style.display === 'flex' ? 'none' : 'flex';
        });

        document.getElementById('btn-load-video').addEventListener('click', () => {
            let input = document.getElementById('video-input').value.trim();
            if (input && visualPlayer) {
                const regex = /(?:youtube\.com\/(?:[^\/]+\/.+\/|(?:v|e(?:mbed)?)\/|.*[?&]v=)|youtu\.be\/)([^"&?\/\s]{11})/i;
                const match = input.match(regex);
                if (match && match[1]) input = match[1];

                currentVideoId = input;
                visualPlayer.loadVideoById({ videoId: currentVideoId });
                audioPlayer.loadVideoById({ videoId: currentVideoId });

                applyGearEffect(0);
                document.getElementById('video-selector').style.display = 'none';
                displaySpeed = 0;
            }
        });

        document.getElementById('btn-unmute').addEventListener('click', () => {
            if (audioPlayer) {
                audioPlayer.unMute(); audioPlayer.setVolume(100);
                document.getElementById('btn-unmute').style.display = 'none';
            }
        });
