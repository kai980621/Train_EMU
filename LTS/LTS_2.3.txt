            { id: 0, name: 'EB', decel: 30.0, color: '#ff3333' }, { id: 1, name: 'B4', decel: 15.0, color: '#ff6633' },
            { id: 2, name: 'B3', decel: 10.0, color: '#ff9933' }, { id: 3, name: 'B2', decel: 5.0, color: '#ffcc33' },
            { id: 4, name: 'B1', decel: 2.5, color: '#ffff33' }, { id: 5, name: 'N', decel: 0.15, color: '#fff' },
            { id: 6, name: 'P1', accel: 1.2, color: '#33ffcc' }, { id: 7, name: 'P2', accel: 1.8, color: '#33ff99' },
            { id: 8, name: 'P3', accel: 2.5, color: '#33ff66' }, { id: 9, name: 'P4', accel: 4.0, color: '#33ff33' },
            { id: 10, name: 'P5', accel: 8.0, color: '#00ff00' }
        ];

        function log(msg, urgent = false) {
            const el = document.getElementById('system-msg');
            el.textContent = msg.toUpperCase();
            el.style.color = urgent ? 'var(--danger)' : 'var(--accent)';
        }

        function onYouTubeIframeAPIReady() {
            apiReady = true;
            const btn = document.getElementById('btn-master-start');
            btn.textContent = 'SYSTEMS ONLINE'; btn.disabled = false;
        }

        function launchSimulator() {
            const playerVars = { 'autoplay': 1, 'controls': 0, 'disablekb': 1, 'fs': 0, 'modestbranding': 1, 'rel': 0, 'showinfo': 0, 'iv_load_policy': 3, 'playsinline': 1, 'enablejsapi': 1 };
            visualPlayer = new YT.Player('visual-player-container', { videoId: currentVideoId, playerVars, events: { 'onReady': onEngineReady, 'onStateChange': onEngineStateChange } });
            audioPlayer = new YT.Player('audio-player-container', { videoId: currentVideoId, playerVars, events: { 'onReady': onEngineReady } });
        }

        let readyCount = 0;
        function onEngineReady() {
            readyCount++;
            if (readyCount >= 2) {
                visualPlayer.mute(); audioPlayer.unMute();
                document.getElementById('startup-screen').style.display = 'none';
                document.getElementById('app-container').style.display = 'block';
                isStarted = true;
                updateMasconUI();
                startEngineLoop();
            }
        }

        function onEngineStateChange(e) { if (e.data === YT.PlayerState.ENDED) visualPlayer.seekTo(0); }

        function startEngineLoop() {
            if (syncCheckInterval) clearInterval(syncCheckInterval);
            refreshStnList();

            syncCheckInterval = setInterval(() => {
                if (!isStarted || isPaused || isSeeking) return;

                const now = new Date();
                document.getElementById('system-clock').textContent = now.toTimeString().split(' ')[0];

                const time = Math.floor(visualPlayer.getCurrentTime());
                const exactTime = visualPlayer.getCurrentTime();

                // Display absolute video seconds for manual editing
                document.getElementById('current-vid-info').textContent = `TIME: ${exactTime.toFixed(1)}s`;

                const vidStations = memory[currentVideoId] || [];
                // Stay at the 'current station' until doors are completely closed + a small buffer
                const currentSTN = vidStations.find(s => exactTime >= s.time - 3 && exactTime <= (s.closeTime || s.time + 30) + 3);

                if (currentSTN) {
                    document.getElementById('stn-name').textContent = currentSTN.name;
                    document.getElementById('stn-dist').textContent = "--- ARRIVED ---";
                    document.getElementById('stn-name').style.color = 'var(--accent)';
                    if (currentSpeed === 0 && pGearLocked && !evalShown) evaluateStop(Math.abs(exactTime - currentSTN.time));
                } else {
                    evalShown = false;
                    const nextSTN = vidStations.find(s => s.time > time);
                    if (nextSTN) {
                        document.getElementById('stn-name').textContent = nextSTN.name;
                        const dist = Math.max(0, Math.floor((nextSTN.time - exactTime) * 16.6) - 1);
                        document.getElementById('stn-dist').textContent = `DIST: ${dist}m | T-${Math.ceil(nextSTN.time - exactTime)}s`;
                        document.getElementById('stn-name').style.color = '#fff';

                        // Broadcast Trigger Logic - Fixed at 15s
                        if (Math.abs((nextSTN.time - exactTime) - 15) < 0.2) {
                            if (platformAnnounce.paused) platformAnnounce.play().catch(() => { });
                        }
                    } else {
                        document.getElementById('stn-name').textContent = "---";
                        document.getElementById('stn-dist').textContent = "EN ROUTE";
                    }
                }

                if (pGearLocked) {
                    const activeStn = vidStations.find(s => Math.abs(s.time - exactTime) < 60);
                    if (activeStn && activeStn.closeTime && activeStn.closeTime > exactTime) {
                        document.getElementById('door-indicator').textContent = `OPEN [ CLOSE IN ${Math.ceil(activeStn.closeTime - exactTime)}s ]`;
                    } else {
                        document.getElementById('door-indicator').textContent = `OPEN [ ${Math.floor(exactTime)}s ]`;
                    }
                } else if (document.getElementById('door-indicator').textContent !== 'DOOR LOCKED') {
                    document.getElementById('door-indicator').textContent = 'DOOR LOCKED';
                }

                if (!isAutoPaused && !visualAutoPaused) {
                    if (visualPlayer.getPlayerState() !== 1) visualPlayer.playVideo();
                    if (audioPlayer.getPlayerState() !== 1) audioPlayer.playVideo();
                }
                if (!isAutoPaused) updatePhysics();
            }, 200);
        }

        function refreshStnList() {
            const container = document.getElementById('stn-list-container');
            const vidStations = memory[currentVideoId] || [];
            if (vidStations.length === 0) {
                container.innerHTML = `<div style="opacity: 0.5; text-align: center; padding: 20px; font-size: 12px;">NO MEMORY DATA</div>`;
                return;
            }
            container.innerHTML = vidStations.map((s, idx) => `
                <div class="stn-row">
                    <input type="text" value="${s.name}" onchange="saveStation(${idx}, 'name', this.value)">
                    <input type="number" value="${Math.floor(s.time)}" onchange="saveStation(${idx}, 'time', this.value)">
                    <input type="number" value="${s.closeTime ? Math.floor(s.closeTime) : ''}" placeholder="--" onchange="saveStation(${idx}, 'closeTime', this.value)">
                    <button class="delete-btn" onclick="deleteStation(${idx})">X</button>
                </div>
            `).join('');
        }

        function saveStation(idx, field, val) {
            const v = (field === 'name') ? val : parseFloat(val);
            memory[currentVideoId][idx][field] = v;
            if (field === 'time') memory[currentVideoId].sort((a, b) => a.time - b.time);
            localStorage.setItem('emu_memory_v2', JSON.stringify(memory));
            refreshStnList();
        }

        function deleteStation(idx) {
            memory[currentVideoId].splice(idx, 1);
            localStorage.setItem('emu_memory_v2', JSON.stringify(memory));
            refreshStnList();
        }

        function clearAllStations() {
            if (confirm("WIPE ALL ROUTE DATA FOR THIS VIDEO?")) {
                memory[currentVideoId] = [];
                localStorage.setItem('emu_memory_v2', JSON.stringify(memory));
                refreshStnList();
            }
        }

        function evaluateStop(diff) {
            evalShown = true;
            const modal = document.getElementById('eval-modal');
            const grade = document.getElementById('eval-grade');
            const diffEl = document.getElementById('eval-diff');

            let text = "POOR", color = "var(--danger)";
            const dist = Math.abs(diff * 16.6).toFixed(1);

            if (diff < 0.3) { text = "EXCELLENT"; color = "gold"; }
            else if (diff < 1.0) { text = "GREAT"; color = "var(--safe)"; }
            else if (diff < 3.0) { text = "GOOD"; color = "var(--accent)"; }

            grade.textContent = text; grade.style.color = color;
            diffEl.textContent = `ERROR: ${dist}m (${diff.toFixed(2)}s)`;
            modal.style.display = 'block';
            setTimeout(() => { modal.style.opacity = '0'; setTimeout(() => { modal.style.display = 'none'; modal.style.opacity = '1'; }, 1000); }, 3000);
        }

        function openJumpMenu(offset) {
            const time = visualPlayer.getCurrentTime();
            const vidStations = memory[currentVideoId] || [];
            let list = [];

            if (offset > 0) {
                document.getElementById('jump-menu-title').textContent = "JUMP TO NEXT";
                list = vidStations.filter(s => s.time > time + 5);
            } else {
                document.getElementById('jump-menu-title').textContent = "JUMP TO PREVIOUS";
                list = [...vidStations].filter(s => s.time < time - 5).reverse();
            }

            const container = document.getElementById('jump-menu-list');
            if (list.length === 0) {
                container.innerHTML = `<div style="opacity: 0.5; text-align: center;">NO STATIONS FOUND</div>`;
            } else {
                container.innerHTML = list.map(s => `
                    <button onclick="executeJump(${s.time})" style="padding: 15px; background: rgba(255,255,255,0.1); border: 1px solid rgba(255,255,255,0.2); color: #fff; font-weight: 800; text-align: left; cursor: pointer; border-radius: 6px; font-size: 16px;">
                        ${s.name} <span style="float: right; opacity: 0.5;">${Math.floor(s.time / 60)}:${Math.floor(s.time % 60).toString().padStart(2, '0')}</span>
                    </button>
                `).join('');
            }
            document.getElementById('jump-menu-modal').style.display = 'block';
        }

        function executeJump(targetTime) {
            document.getElementById('jump-menu-modal').style.display = 'none';
            currentSpeed = 0; applyGear(5); // Stop physics before jumping
            visualPlayer.seekTo(targetTime, true);
            audioPlayer.seekTo(targetTime, true);
            log(`JUMPED TO STATION`);
        }

        function importBatch() {
            try {
                const data = JSON.parse(document.getElementById('batch-input').value);
                if (Array.isArray(data)) {
                    memory[currentVideoId] = data;
                    localStorage.setItem('emu_memory_v2', JSON.stringify(memory));
                    refreshStnList();
                    log("ROUTE IMPORTED SUCCESSFULLY");
                }
            } catch (e) { log("IMPORT FAILED: INVALID JSON", true); }
        }

        function updatePhysics() {
            const gear = GEARS[currentGearId];
            let delta = (gear.accel || 0) - (gear.decel || 0);
            if (currentGearId === 5) delta = -0.15; // Natural Friction

            currentSpeed += delta * 0.2;
            if (currentSpeed < 0.1) currentSpeed = 0;
            currentSpeed = Math.min(140, currentSpeed);

            if (visualPlayer && visualPlayer.setPlaybackRate) {
                let rate = pGearLocked ? 1.0 : (0.5 + currentSpeed * 0.025);
                rate = Math.max(0.5, Math.min(2.0, Math.round(rate * 10) / 10));
                if (rate !== lastSetRate) {
                    visualPlayer.setPlaybackRate(rate);
                    lastSetRate = rate;
                }
            }

            document.getElementById('speed-value').textContent = Math.floor(currentSpeed);
            document.getElementById('speed-value').style.color = gear.color;

            // Power-Save Auto Pause
            if (currentSpeed === 0 && !pGearLocked && !isAutoPaused && !visualAutoPaused && !doorTimer) {
                doorTimer = setTimeout(() => {
                    if (currentSpeed === 0 && !pGearLocked) {
                        visualAutoPaused = true; visualPlayer.pauseVideo();
                        log("STOPPED - STANDBY MODE", true);
                    }
                }, 5000);
                audioTimer = setTimeout(() => {
                    if (currentSpeed === 0 && !pGearLocked) {
                        isAutoPaused = true; audioPlayer.pauseVideo();
                        log("STOPPED - FULL SHUTDOWN", true);
                    }
                }, 10000);
            }
        }

        function memorizeStation() {
            const name = document.getElementById('stn-input').value.trim() || 'STATION';
            const exactTime = visualPlayer.getCurrentTime();

            if (!memory[currentVideoId]) memory[currentVideoId] = [];
            memory[currentVideoId].push({ name, time: exactTime, closeTime: exactTime + 20 });
            memory[currentVideoId].sort((a, b) => a.time - b.time);
            localStorage.setItem('emu_memory_v2', JSON.stringify(memory));
            document.getElementById('stn-input').value = "";
            refreshStnList();
            hideSmartToast(); log(`${name} MEMORIZED AND UPDATED`);
            if (navigator.vibrate) navigator.vibrate([40, 40, 40]);
        }

        function hideSmartToast() { document.getElementById('smart-toast').style.display = 'none'; }

        function applyGear(idx) {
            currentGearId = idx;
            if (navigator.vibrate) navigator.vibrate(20);
            updateMasconUI();
            if (currentGearId > 5) {
                clearTimeout(doorTimer); clearTimeout(audioTimer);
                doorTimer = null; audioTimer = null;
                isAutoPaused = false; visualAutoPaused = false;
                visualPlayer.playVideo(); audioPlayer.playVideo();
                log("DEPARTURE INITIALIZED");
            }
        }

        function updateMasconUI() {
            const gear = GEARS[currentGearId];
            const handle = document.getElementById('mascon-handle');
            handle.style.top = `${((10 - currentGearId) / 10) * 100}%`;
            handle.textContent = gear.name;
            document.getElementById('gear-indicator').textContent = gear.name;
            document.getElementById('gear-indicator').style.color = gear.color;
        }

        function handleAction(code) {
            if (!isStarted || isPaused) return;
            const ind = document.getElementById('door-indicator');

            // Single Button UI toggles doors dynamically
            if (code === 'KeyE' || code === 'KeyQ' || code === 'ToggleDoors' || code === 'Space') {
                if (currentSpeed > 0) return; // Prevent door operations while moving

                // CLOSE DOORS
                if (code === 'KeyE' || ((code === 'ToggleDoors' || code === 'Space') && pGearLocked)) {
                    doorCloseSFX.play().catch(() => { });
                    pGearLocked = false;
                    ind.className = ''; ind.textContent = 'DOOR LOCKED';
                    log("READY FOR DEPARTURE");

                    const exactTime = visualPlayer.getCurrentTime();
                    if (Math.abs(exactTime - audioPlayer.getCurrentTime()) > 0.5) {
                        audioPlayer.seekTo(exactTime, true);
                        log("A/V RESYNCED");
                    }

                    // Auto-Update Close Time For Active Station
                    const stn = memory[currentVideoId]?.find(s => Math.abs(s.time - exactTime) < 60);
                    if (stn) {
                        stn.closeTime = exactTime;
                        localStorage.setItem('emu_memory_v2', JSON.stringify(memory));
                        refreshStnList();
                        log(`${stn.name} CLOSE TIME UPDATED`);
                    }
                }
                // OPEN DOORS
                else if (code === 'KeyQ' || ((code === 'ToggleDoors' || code === 'Space') && !pGearLocked)) {
                    doorOpenSFX.play().catch(() => { });
                    pGearLocked = true;
                    ind.className = 'door-open'; ind.textContent = 'DOOR OPEN';
                    if (currentGearId > 5) applyGear(5);
                    log("DOOR OPERATIONS ACTIVE");

                    const exactTime = visualPlayer.getCurrentTime();
                    if (Math.abs(exactTime - audioPlayer.getCurrentTime()) > 0.5) {
                        audioPlayer.seekTo(exactTime, true);
                        log("A/V RESYNCED");
                    }

                    // Auto-Update Open Time OR Prompt for New Station
                    const stn = memory[currentVideoId]?.find(s => Math.abs(s.time - exactTime) < 40);
                    if (stn) {
                        stn.time = exactTime;
                        localStorage.setItem('emu_memory_v2', JSON.stringify(memory));
                        refreshStnList();
                        log(`${stn.name} OPEN TIME UPDATED`);
                    } else {
                        document.getElementById('smart-toast').style.display = 'block';
                    }
                }
            } else if (code === 'KeyZ') {
                if (pGearLocked) log("CLEAR DOORS BEFORE START", true);
                else applyGear(Math.min(10, currentGearId + 1));
            } else if (code === 'KeyS') {
                applyGear(Math.max(0, currentGearId - 1));
            } else if (code === 'KeyA') {
                applyGear(5);
            } else if (code === 'KeyX') {
                audioPlayer.seekTo(visualPlayer.getCurrentTime(), true);
                log("AV SYNC ALIGNED");
            }
        }

        let pendingVideoId = null;
        document.getElementById('btn-check-title').addEventListener('click', async () => {
            let val = document.getElementById('video-input').value.trim();
            const m = val.match(/(?:v=|\/)([0-9A-Za-z_-]{11})/);
            if (!m) { log("INVALID URL", true); return; }
            const vid = m[1];

            document.getElementById('btn-check-title').textContent = "FETCHING...";
            try {
                // Fetch from oEmbed
                const response = await fetch(`https://www.youtube.com/oembed?url=https://www.youtube.com/watch?v=${vid}&format=json`);
                const data = await response.json();
                pendingVideoId = vid;
                document.getElementById('video-check-title').textContent = data.title;
                document.getElementById('video-info-preview').style.display = 'block';
                document.getElementById('btn-load-link').style.display = 'block';
                document.getElementById('btn-check-title').style.display = 'none';
                log("VIDEO VERIFIED");
            } catch (e) {
                log("FETCH FAILED", true);
                document.getElementById('btn-check-title').textContent = "FETCH VIDEO INFO";
            }
        });

        document.getElementById('btn-load-link').addEventListener('click', () => {
            if (!pendingVideoId) return;
            currentVideoId = pendingVideoId;
            visualPlayer.loadVideoById(currentVideoId);
            audioPlayer.loadVideoById(currentVideoId);
            currentSpeed = 0; applyGear(5);
            evalShown = false; stopStartTime = null;
            document.getElementById('video-overlay').style.display = 'none';
            document.getElementById('video-info-preview').style.display = 'none';
            document.getElementById('btn-load-link').style.display = 'none';
            document.getElementById('btn-check-title').style.display = 'block';
            document.getElementById('btn-check-title').textContent = "FETCH VIDEO INFO";
            refreshStnList(); log("DATA LINK REFRESHED");
        });

        // Mascon Interactivity
        const track = document.getElementById('mascon-track');
        track.addEventListener('pointerdown', (e) => {
            const move = (ev) => {
                const rect = track.getBoundingClientRect();
                const p = Math.max(0, Math.min(1, (ev.clientY - rect.top) / rect.height));
                applyGear(10 - Math.round(p * 10));
            };
            move(e);
            window.addEventListener('pointermove', move);
            window.addEventListener('pointerup', () => window.removeEventListener('pointermove', move), { once: true });
        });
